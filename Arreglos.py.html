28       if n % 2 == 0 or n % 3 == 0:

139  def suma_digitos(n):
140      """Devuelve la suma de los dígitos del número entero n."""
141      return sum(int(digit) for digit in str(abs(n)))
142  
143  # Inicializar un arreglo vacìo
144  numeros = []
145  print("Introduce 10 números enteros:")
146  
147  for i in range(10):
148      num = int(input(f"Número {i + 1}: "))
149      numeros.append(num)
150  
151  # Nùmero- suma de dìgitos
152  max_suma_digitos = -1
153  posicion_max = -1
154  
155  for i in range(len(numeros)):
156      suma_actual = suma_digitos(numeros[i])
157      if suma_actual > max_suma_digitos:
158          max_suma_digitos = suma_actual
159          posicion_max = i
160  
161  # Salida
162  print(f"El número con la mayor suma de dígitos es {numeros[posicion_max]} en la posición {posicion_max}.")

209  sexos = []

214  N = int(input("Ingrese el número de estudiantes (N <= 50): "))

216  # Verificar que el número de estudiantes sea válido
217  if N > MAX_ESTUDIANTES or N <= 0:
218      print("Número de estudiantes no válido.")
219  else:
220      # Leer los datos de los estudiantes
221      for i in range(N):
222          print(f"\nEstudiante {i + 1}:")
223          cedula = input("Cédula: ")
224          sexo = int(input("Sexo (1: masculino, 2: femenino): "))
225          sueldo = int(input("Sueldo: "))
226          trabaja_estudiante = int(input("Trabaja (1: sí, 2: no): "))
227  
228          # Almacenar los datos en los arreglos

230          sexos.append(sexo)

236      total_mujeres = sexos.count(2)
237  
238      if total_hombres > 0:

245          porcentaje_hombres_trabajan = 0
246          promedio_sueldo_hombres_trabajan = 0
247  
248      if total_mujeres > 0:
249          total_trabajan_mujeres = sum(1 for i in range(N) if sexos[i] == 2 and trabaja[i] == 1)
250          total_sueldo_mujeres_trabajan = sum(sueldos[i] for i in range(N) if sexos[i] == 2 and trabaja[i] == 1)
251          porcentaje_mujeres_trabajan = (total_trabajan_mujeres / total_mujeres) * 100
252          promedio_sueldo_mujeres_trabajan = (
